  <style>

    body { font-family: Arial; padding: 20px; }

    table { border-collapse: collapse; margin-top: 20px; }

  </style>

<body>

  <h2>Calculadora de Ahorro Mensual</h2>

  <div id="formulario"></div>

  <button onclick="calcular()">Calcular</button>

  <div id="resultado"></div>

 

  <script>

    const meses = ["Septiembre", "Octubre", "Noviembre", "Diciembre"];

    const ingresos = [1874000, 2774000, 1874000, 2774000];

    const arriendos = [400000, 650000, 400000, 650000];

 

    let formulario = document.getElementById("formulario");

    meses.forEach((mes, i) => {

      formulario.innerHTML += `<h3>${mes}</h3>

        Gastos comunes: <input type="number" id="gcom${i}" value="0"><br>

        Gastos adicionales: <input type="number" id="gadd${i}" value="0"><br>

        Ingresos adicionales: <input type="number" id="ingadd${i}" value="0"><br>`;

    });

 

    function calcular() {

      let resultado = "<table><tr><th>Mes</th><th>Ingreso</th><th>Gastos</th><th>Saldo</th><th>Ahorro</th></tr>";

      let ahorroTotal = 0;

 

      meses.forEach((mes, i) => {

        let ingreso = ingresos[i] + Number(document.getElementById(ingadd${i}).value);

        let gastos = arriendos[i] + Number(document.getElementById(gcom${i}).value) + Number(document.getElementById(gadd${i}).value);

        let saldo = ingreso - gastos;

        let ahorro = saldo > 0 ? saldo : 0;

        ahorroTotal += ahorro;

 

        resultado += <tr><td>${mes}</td><td>${ingreso}</td><td>${gastos}</td><td>${saldo}</td><td>${ahorro}</td></tr>;

      });

 

      resultado += <tr><td colspan="4"><strong>Ahorro total hasta diciembre</strong></td><td><strong>${ahorroTotal}</strong></td></tr></table>;

      document.getElementById("resultado").innerHTML = resultado;

    }

  </script>

</body>
